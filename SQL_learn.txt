## Exercise 9 = 
Query the two cities in STATION with the SHORTEST and LONGEST CITY names
as well as their respective LENGTHS (i.e.: number of characters in the name). 
If there is more than one smallest or largest city, choose the one that comes first when ordered alphabetically.
    - use LENGTH() ==> count char length 
    - ORDER BY () ASC ==> orders the () from small to big 

    SELECT CITY, LENGTH(CITY)
    FROM STATION
    ORDER BY LENGTH(CITY) ASC, CITY ASC
    LIMIT 1

    UNION # combines the result into one output 

    SELECT CITY, LENGTH(CITY)
    FROM STATION
    ORDER BY LENGTH(CITY) DESC, CITY ASC
    LIMIT 1;
    # solution doesnt work because of 'LIMIT' keyword wont work 
    # correct solution using F = 
    SELECT CITY, LENGTH(CITY)
    FROM (
        SELECT CITY, LENGTH(CITY)
        FROM STATION
        ORDER BY LENGTH(CITY) ASC, CITY ASC
        FETCH FIRST 1 ROW ONLY
    )

    UNION

    SELECT CITY, LENGTH(CITY)
    FROM (
        SELECT CITY, LENGTH(CITY)
        FROM STATION
        ORDER BY LENGTH(CITY) DESC, CITY ASC
        FETCH FIRST 1 ROW ONLY
    );

## Exercise 10 -- using IF with multiple conditions
multiple conditions use CASE -- failed attempt 
SELECT CITY,
    CASE 
        WHEN STARTS_WITH(CITY, 'a') OR STARTS_WITH(CITY, 'i') OR STARTS_WITH(CITY, 'u') OR STARTS_WITH(CITY, 'e') OR STARTS_WITH(CITY, 'o')
    END AS Status
FROM STATION;
# correct solution = use regular "or" statements 
SELECT DISTINCT CITY # distinct to get unique elements 
FROM STATION
WHERE CITY LIKE 'A%' # A% means start with A and ends with whatever char
   OR CITY LIKE 'E%'
   OR CITY LIKE 'I%'
   OR CITY LIKE 'O%'
   OR CITY LIKE 'U%';
